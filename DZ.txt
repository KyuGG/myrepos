struct car
{
	int id;
	string mark;
	string color;
	string categoty;
	int year;
	float price;
	bool old;
};

vector<car> toprogram(string name)
{
	ifstream fin;
	fin.open(name);
	vector<car> v;
	if (fin.is_open() == true)
	{
		while (!fin.eof())
		{
			car buf;
			fin >> buf.id;
			fin >> buf.mark;
			fin >> buf.year;
			fin >> buf.color;
			fin >> buf.categoty;
			fin >> buf.price;
			fin >> buf.old;
			v.push_back(buf);
		}
		return v;
		fin.close();
	}
	else
	{
		cout << "Не удалось открыть файл" << endl;
	}
}

void withoutcar(vector<car> &v, int id)
{
	for (int i = 0; i < v.size(); i++)
	{
		if (v[i].id == id)
		{
			v.erase(v.begin() + i);
		}
	}
}
